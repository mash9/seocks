ALTER TABLE CODE
 DROP PRIMARY KEY CASCADE;

DROP TABLE CODE CASCADE CONSTRAINTS;

CREATE TABLE CODE
(
  TYPE  VARCHAR2(50 BYTE),
  CODE  VARCHAR2(50 BYTE),
  NAME  VARCHAR2(50 BYTE),
  ORD   INTEGER
);


CREATE UNIQUE INDEX CODE_PK ON CODE
(TYPE, CODE);


ALTER TABLE CODE ADD (
  CONSTRAINT CODE_PK
  PRIMARY KEY
  (TYPE, CODE)
  USING INDEX CODE_PK
  ENABLE VALIDATE);
