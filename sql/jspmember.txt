ALTER TABLE JSPMEMBER
 DROP PRIMARY KEY CASCADE;

DROP TABLE JSPMEMBER CASCADE CONSTRAINTS;

CREATE TABLE JSPMEMBER
(
  ID         VARCHAR2(100 BYTE)                 NOT NULL,
  PASS       VARCHAR2(100 BYTE),
  NAME       VARCHAR2(100 BYTE),
  EMAIL      VARCHAR2(100 BYTE),
  MTEMP      VARCHAR2(100 BYTE),
  MADDRESS1  VARCHAR2(500 BYTE),
  MADDRESS2  VARCHAR2(500 BYTE),
  GENDER     VARCHAR2(100 BYTE),
  BIRTHDAY   VARCHAR2(100 BYTE),
  INFO       VARCHAR2(1000 BYTE),
  PHONE      VARCHAR2(100 BYTE)
);

COMMENT ON TABLE JSPMEMBER IS '회원DB';

COMMENT ON COLUMN JSPMEMBER.ID IS '아이디';

COMMENT ON COLUMN JSPMEMBER.PASS IS '비밀번호';

COMMENT ON COLUMN JSPMEMBER.NAME IS '이름';

COMMENT ON COLUMN JSPMEMBER.EMAIL IS '이메일';

COMMENT ON COLUMN JSPMEMBER.MTEMP IS '우편번호';

COMMENT ON COLUMN JSPMEMBER.MADDRESS1 IS '주소1';

COMMENT ON COLUMN JSPMEMBER.MADDRESS2 IS '주소2';

COMMENT ON COLUMN JSPMEMBER.GENDER IS '성별';

COMMENT ON COLUMN JSPMEMBER.BIRTHDAY IS '생년월일';

COMMENT ON COLUMN JSPMEMBER.INFO IS '자기소개';

COMMENT ON COLUMN JSPMEMBER.PHONE IS '연락처';



CREATE UNIQUE INDEX JSPMEMBER_PK ON JSPMEMBER(ID);

ALTER TABLE JSPMEMBER ADD (
  CONSTRAINT JSPMEMBER_PK
  PRIMARY KEY
  (ID)
  USING INDEX JSPMEMBER_PK
  ENABLE VALIDATE);
