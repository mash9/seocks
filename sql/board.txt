DROP TABLE BOARD CASCADE CONSTRAINTS;

CREATE TABLE BOARD
(
  NUM        INTEGER                            NOT NULL,
  SUBJECT    VARCHAR2(500 BYTE),
  WRITER     VARCHAR2(100 BYTE),
  EMAIL      VARCHAR2(500 BYTE),
  PASSWORD   VARCHAR2(100 BYTE),
  REG_DATE   DATE,
  REF        INTEGER,
  RE_LEVEL   INTEGER,
  RE_STEP    INTEGER,
  READCOUNT  INTEGER,
  LIKECOUNT  INTEGER,
  CONTENTS   VARCHAR2(2000 BYTE)
);

COMMENT ON TABLE BOARD IS '공지게시판DB';

COMMENT ON COLUMN BOARD.NUM IS '테이블번호';

COMMENT ON COLUMN BOARD.SUBJECT IS '제목';

COMMENT ON COLUMN BOARD.WRITER IS '아이디';

COMMENT ON COLUMN BOARD.EMAIL IS '이메일';

COMMENT ON COLUMN BOARD.PASSWORD IS '비밀번호';

COMMENT ON COLUMN BOARD.REG_DATE IS '작성일';

COMMENT ON COLUMN BOARD.REF IS '글레벨';

COMMENT ON COLUMN BOARD.RE_LEVEL IS '리플레벨';

COMMENT ON COLUMN BOARD.RE_STEP IS '리리플레벨';

COMMENT ON COLUMN BOARD.READCOUNT IS '조회수';

COMMENT ON COLUMN BOARD.LIKECOUNT IS '좋아요';

COMMENT ON COLUMN BOARD.CONTENTS IS '내용';
