DROP TABLE C##SHOPDB.BOARD CASCADE CONSTRAINTS;

CREATE TABLE C##SHOPDB.BOARD
(
  NUM        INTEGER                            NOT NULL,
  SUBJECT    VARCHAR2(500 BYTE),
  WRITER     VARCHAR2(100 BYTE),
  EMAIL      VARCHAR2(500 BYTE),
  PASSWORD   VARCHAR2(100 BYTE),
  REG_DATE   DATE,
  REF        INTEGER,
  RE_LEVEL   INTEGER,
  RE_STEP    INTEGER,
  READCOUNT  INTEGER,
  LIKECOUNT  INTEGER,
  CONTENTS   VARCHAR2(2000 BYTE)
)
TABLESPACE SHOP
RESULT_CACHE (MODE DEFAULT)
PCTUSED    40
PCTFREE    10
INITRANS   1
MAXTRANS   255
STORAGE    (
            INITIAL          64K
            NEXT             1M
            MAXSIZE          UNLIMITED
            MINEXTENTS       1
            MAXEXTENTS       UNLIMITED
            PCTINCREASE      0
            FREELISTS        1
            FREELIST GROUPS  1
            BUFFER_POOL      DEFAULT
            FLASH_CACHE      DEFAULT
            CELL_FLASH_CACHE DEFAULT
           )
LOGGING
NOCOMPRESS
NOCACHE
NOPARALLEL
MONITORING;

COMMENT ON TABLE C##SHOPDB.BOARD IS '공지게시판DB';

COMMENT ON COLUMN C##SHOPDB.BOARD.SUBJECT IS '제목';

COMMENT ON COLUMN C##SHOPDB.BOARD.WRITER IS '아이디';

COMMENT ON COLUMN C##SHOPDB.BOARD.EMAIL IS '이메일';

COMMENT ON COLUMN C##SHOPDB.BOARD.PASSWORD IS '비밀번호';

COMMENT ON COLUMN C##SHOPDB.BOARD.REG_DATE IS '작성일';

COMMENT ON COLUMN C##SHOPDB.BOARD.REF IS '글레벨';

COMMENT ON COLUMN C##SHOPDB.BOARD.RE_LEVEL IS '리플레벨';

COMMENT ON COLUMN C##SHOPDB.BOARD.RE_STEP IS '리리플레벨';

COMMENT ON COLUMN C##SHOPDB.BOARD.READCOUNT IS '조회수';
