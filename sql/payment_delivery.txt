ALTER TABLE PAYMENT_DELIVERY
 DROP PRIMARY KEY CASCADE;

DROP TABLE PAYMENT_DELIVERY CASCADE CONSTRAINTS;

CREATE TABLE PAYMENT_DELIVERY
(
  PAY_ID       VARCHAR2(100 BYTE)               NOT NULL,
  USER_ID      VARCHAR2(100 BYTE),
  NAME         VARCHAR2(100 BYTE),
  POST_NUMBER  VARCHAR2(100 BYTE),
  ADDRESS1     VARCHAR2(500 BYTE),
  ADDRESS2     VARCHAR2(500 BYTE),
  PHONE        VARCHAR2(100 BYTE),
  PAY_TYPE     VARCHAR2(100 BYTE),
  MESSAGE      VARCHAR2(1000 BYTE),
  BUY_DATE     DATE
);

COMMENT ON TABLE PAYMENT_DELIVERY IS '배송정보';

COMMENT ON COLUMN PAYMENT_DELIVERY.PAY_ID IS '결제 ID';

COMMENT ON COLUMN PAYMENT_DELIVERY.USER_ID IS '결제자 ID';

COMMENT ON COLUMN PAYMENT_DELIVERY.NAME IS '이름';

COMMENT ON COLUMN PAYMENT_DELIVERY.POST_NUMBER IS '우편번호';

COMMENT ON COLUMN PAYMENT_DELIVERY.ADDRESS1 IS '주소1';

COMMENT ON COLUMN PAYMENT_DELIVERY.ADDRESS2 IS '주소2';

COMMENT ON COLUMN PAYMENT_DELIVERY.PHONE IS '휴대번호';

COMMENT ON COLUMN PAYMENT_DELIVERY.PAY_TYPE IS '결제방법';

COMMENT ON COLUMN PAYMENT_DELIVERY.MESSAGE IS '기타 메세지';



CREATE UNIQUE INDEX PAYMENT_DELIVERY_PK ON PAYMENT_DELIVERY
(PAY_ID);


ALTER TABLE PAYMENT_DELIVERY ADD (
  CONSTRAINT PAYMENT_DELIVERY_PK
  PRIMARY KEY
  (PAY_ID)
  USING INDEX PAYMENT_DELIVERY_PK
  ENABLE VALIDATE);
