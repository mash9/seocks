ALTER TABLE SHOPPINGITEM
 DROP PRIMARY KEY CASCADE;

DROP TABLE SHOPPINGITEM CASCADE CONSTRAINTS;

CREATE TABLE SHOPPINGITEM
(
  ITEMBUYNO   INTEGER                           NOT NULL,
  PNO         VARCHAR2(100 BYTE),
  PNAME       VARCHAR2(500 BYTE),
  PPRICE      INTEGER,
  PMAINIMG    VARCHAR2(500 BYTE),
  PCATE       INTEGER,
  PSIZE       VARCHAR2(100 BYTE),
  QTY         INTEGER,
  TOTALPRICE  INTEGER,
  REG_DATE    DATE,
  TEMP        VARCHAR2(2000 BYTE)
);

COMMENT ON TABLE SHOPPINGITEM IS '장바구니DB';

COMMENT ON COLUMN SHOPPINGITEM.ITEMBUYNO IS '구매번호';

COMMENT ON COLUMN SHOPPINGITEM.PNO IS '상품번호';

COMMENT ON COLUMN SHOPPINGITEM.PNAME IS '상품이름';

COMMENT ON COLUMN SHOPPINGITEM.PPRICE IS '상품가격';

COMMENT ON COLUMN SHOPPINGITEM.PMAINIMG IS '상품메인이미지';

COMMENT ON COLUMN SHOPPINGITEM.PCATE IS '구분';

COMMENT ON COLUMN SHOPPINGITEM.PSIZE IS '사이즈';

COMMENT ON COLUMN SHOPPINGITEM.QTY IS '수량';

COMMENT ON COLUMN SHOPPINGITEM.TOTALPRICE IS '총가격';

COMMENT ON COLUMN SHOPPINGITEM.REG_DATE IS '구매일';

COMMENT ON COLUMN SHOPPINGITEM.TEMP IS '임시';



CREATE UNIQUE INDEX SHOPPINGITEM_PK ON SHOPPINGITEM
(ITEMBUYNO);


ALTER TABLE SHOPPINGITEM ADD (
  CONSTRAINT SHOPPINGITEM_PK
  PRIMARY KEY
  (ITEMBUYNO)
  USING INDEX SHOPPINGITEM_PK
  ENABLE VALIDATE);
